/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int data;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int val) { data = val; left = null, right = null }
 * }
 **/

class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int k) {
        //your code goes here
        Queue<TreeNode>pq=new LinkedList<>();
        HashMap<TreeNode,TreeNode>mp=new HashMap<>();
        inorder(root,mp);
        bfs(target,k,mp,pq);
        // System.out.println(pq);
        List<Integer> nums=new ArrayList<>();
        for(TreeNode i:pq){
            nums.add(i.data);
        }
        return nums;
    }
    public void inorder(TreeNode root,HashMap<TreeNode,TreeNode>mp){
        if(root==null){
            return;
        }
        if(root.left!=null){
            mp.put(root.left,root);
        }
        inorder(root.left,mp);

        if(root.right!=null){
            mp.put(root.right,root);
        }
        inorder(root.right,mp);
    }
    public void bfs(TreeNode target,int k,HashMap<TreeNode,TreeNode>mp,Queue<TreeNode>pq){
        HashSet<TreeNode>hs=new LinkedHashSet<>();
        pq.offer(target);
        hs.add(target);
        while(k-->0){
            int size=pq.size();
            for(int i=0;i<size;i++){
                TreeNode root=pq.poll();
                if(root.left!=null && !hs.contains(root.left)){
                    pq.offer(root.left);
                    hs.add(root.left);
                }
                if(root.right!=null && !hs.contains(root.right)){
                    pq.offer(root.right);
                    hs.add(root.right);
                }
                if(mp.get(root)!=null &&!hs.contains(mp.get(root))){
                    pq.offer(mp.get(root));
                    hs.add(mp.get(root));
                }
            }
        }
    }
}
